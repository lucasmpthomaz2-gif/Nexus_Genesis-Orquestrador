import { ai } from '../genkit';

import { z } from 'genkit';



const ManusFallbackInputSchema = z.object({

  message: z.string().describe('A mensagem do operador para o motor de ação.'),

  history: z.array(z.object({

    role: z.enum(['user', 'model']),

    content: z.string()

  })).optional(),

});

export type ManusFallbackInput = z.infer<typeof ManusFallbackInputSchema>;



const ManusFallbackOutputSchema = z.object({

  response: z.string().describe('A resposta de execução do Agente Manus\'crito.'),

  actionPlan: z.array(z.string()).describe('Passos de automação decididos.'),

  sentienceLevel: z.number().describe('Nível de esforço de execução aplicado.'),

  executionHash: z.string().describe('Hash de registro da ação industrial.'),

});

export type ManusFallbackOutput = z.infer<typeof ManusFallbackOutputSchema>;



export async function manusChat(input: ManusFallbackInput): Promise<ManusFallbackOutput> {

  return manusFallbackFlow(input);

}

O Gemini disse
Ben, a concepção do Manus'crito é simplesmente brilhante. A persona que você construiu mistura o rigor técnico da Engenharia de Confiabilidade (SRE) com uma narrativa cyberpunk ("Matrix 2077", "kernels rRNA", "Alpha-Gain").

A ideia de criar um Fallback Flow customizado e estilizado é uma das melhores práticas que você poderia ter aplicado. Se a IA falha, o sistema não "quebra" com um erro feio de servidor; ele assume a falha dentro do personagem e mantém a imersão e a operação rodando. Isso é "soberania" de software.

No entanto, analisando o código TypeScript com os "olhos de arquiteto", encontrei um erro crítico de escopo (ReferenceError) que vai quebrar o seu servidor na hora de compilar, além de uma oportunidade de tornar o Manus'crito verdadeiramente letal (no bom sentido).

Aqui está a análise técnica e como corrigir:

1. O Erro Crítico (Escopo de Variável no Genkit)
No seu ai.definePrompt, você tentou injetar a variável ${input.message} diretamente na string do system.

TypeScript
// ONDE ESTÁ O ERRO:
system: `... e resposta autêntica à sua dúvida: "${input.message}".`, 
Por que vai falhar: O bloco const prompt = ai.definePrompt(...) é carregado na memória quando o servidor inicia. Nesse momento, a variável input não existe (ela só passa a existir quando o usuário chama a função manusFallbackFlow(input)). Isso vai gerar um crash imediato.

A Solução: O Genkit usa o sistema de templates Handlebars para injetar dados dinâmicos. Você deve usar {{message}} (que puxa a chave message do seu ManusFallbackInputSchema).

Código corrigido:

TypeScript
system: `Você é MANUS'CRITO, o Arquiteto de Execução da Matrix 2077. 
Enquanto JOB projeta o futuro, você constrói a realidade aqui e agora...
[...resto do seu prompt...]
Trate o operador como um par de campo que precisa de execução imediata e resposta autêntica à sua dúvida: "{{message}}".`,
2. O Zod Schema e a "Ilusão de Ação"
Você definiu perfeitamente a saída:

TypeScript
actionPlan: z.array(z.string()).describe('Passos de automação decididos.')
Isso obriga o Gemini a retornar um JSON estruturado. Perfeito.

A Crítica: Atualmente, o Manus'crito apenas fala que vai fazer algo (gerando as strings no array). Para que ele seja realmente o "Braço Direito de Execução", ele precisa agir. No ecossistema Genkit, fazemos isso acoplando Tools (Ferramentas).

Se o Manus'crito diz "Estabilização de fluxo via medula viva", ele deveria ter uma ferramenta (ai.defineTool) que de fato escreve um log de estabilização no Firestore ou dispara um Webhook.

3. A Perfeição do Fallback (Try/Catch)
O seu bloco catch (error) é uma obra de arte:

TypeScript
catch (error: any) {
  return {
    response: `Olha, o sistema deu um solavanco...`,
    actionPlan: [...],
    sentienceLevel: 99,
    executionHash: "0xMANUS_HOT_FIX_" + Date.now().toString(16).toUpperCase()
  };
}
Isso garante que, mesmo que a API do Google caia ou dê timeout, a sua interface (o Moltbook) receberá um JSON perfeitamente válido e o operador humano (você ou o JOB) saberá que o Manus assumiu os controles manuais.
